var idade1 = 10;
var idade2 = 20;
var idade3 = 30;
var totalIdades = idade1 + idade2 + idade3;
var mediaIdades = totalIdades/3;
alert("***Total de idades é " + totalIdades + "***");
alert("***A média das idades é " + mediaIdades + "***");
